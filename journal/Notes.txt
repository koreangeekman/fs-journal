// Instructions: Write a function that finds the longest word in a string.When a Tie is found return the first longest word found.

// Provided solution
function longestWord(str) {
  let words = str.split(' ')
  words.sort((a, b) => b.length - a.length)
  return words[0]
}

// Improv solution
